// --- 1. グローバル状態管理 ---
let state = {
    thick: 5,
    colorDmg: 0,
    brand: 'milbon',
    capturedPhoto: null,
    currentLevel: 'M'
};

// 薬剤メーカー別プリセット (1剤:水:油)
const brandPresets = {
    milbon:  { AX:[90,5,5], SH:[80,10,10], H:[70,20,10], M:[50,40,10], L:[30,60,10], B:[10,80,10] },
    arimino: { AX:[100,0,0], SH:[85,10,5],  H:[70,20,10], M:[50,40,10], L:[25,65,10], B:[5,85,10] },
    wella:   { AX:[85,10,5], SH:[75,15,10], H:[65,25,10], M:[55,35,10], L:[45,45,10], B:[35,55,10] },
    custom:  { AX:[80,10,10], SH:[70,15,15], H:[60,20,20], M:[50,30,20], L:[40,40,20], B:[30,50,20] }
};

// --- 2. UI操作・診断関数 ---
function updateState(el, key, val) {
    el.parentNode.querySelectorAll('button').forEach(b => b.classList.remove('active'));
    el.classList.add('active');
    state[key] = val;
}

function generatePlan() {
    const { thick, colorDmg, brand } = state;
    const score = (thick + 10) - colorDmg; // 簡易ロジック

    let lv = 'B';
    if(score >= 18) lv = 'AX';
    else if(score >= 15) lv = 'SH';
    else if(score >= 12) lv = 'H';
    else if(score >= 9)  lv = 'M';
    else if(score >= 6)  lv = 'L';
    state.currentLevel = lv;

    const formula = brandPresets[brand][lv];
    const total = 120;
    const m = Math.round(total * formula[0]/100);
    const w = Math.round(total * formula[1]/100);
    const o = Math.round(total * formula[2]/100);

    let risks = [];
    if(colorDmg >= 6) risks.push("⚠️ ハイダメージ：前処理剤を併用してください");
    if(lv === 'L' || lv === 'B') risks.push("⚠️ 軟化が速い可能性：5分おきにチェック");

    document.getElementById('plan-output').innerHTML = `
        <div class="card" style="border-left:5px solid var(--primary)">
            <h3>推奨プラン: ${lv}ランク</h3>
            <div style="display:flex; justify-content:space-between; text-align:center;">
                <div>1剤<br><strong>${m}g</strong></div>
                <div>水/減力<br><strong>${w}g</strong></div>
                <div>保護剤<br><strong>${o}g</strong></div>
            </div>
            ${risks.length ? `<div class="risk-box">${risks.join('<br>')}</div>` : ''}
        </div>
    `;
}

// --- 3. カメラ機能 ---
let stream = null;
async function toggleCamera() {
    const video = document.getElementById('video');
    const btn = document.getElementById('cam-btn');
    const snapBtn = document.getElementById('snap-btn');

    if (!stream) {
        stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } });
        video.srcObject = stream;
        video.style.display = 'block';
        btn.innerText = "カメラを止める";
        snapBtn.style.display = 'block';
    } else {
        stream.getTracks().forEach(track => track.stop());
        stream = null;
        video.style.display = 'none';
        btn.innerText = "カメラを起動";
        snapBtn.style.display = 'none';
    }
}

function takePhoto() {
    const video = document.getElementById('video');
    const canvas = document.getElementById('canvas');
    canvas.width = video.videoWidth;
    canvas.height = video.videoHeight;
    canvas.getContext('2d').drawImage(video, 0, 0);
    state.capturedPhoto = canvas.toDataURL('image/png');
    document.getElementById('photo-preview').innerHTML = `<img src="${state.capturedPhoto}">`;
    alert("画像を解析用データとして一時保存しました");
}

// --- 4. タイマー & ユーティリティ ---
let timer;
function startTimer(sec) {
    clearInterval(timer);
    let s = sec;
    timer = setInterval(() => {
        s--;
        const m = Math.floor(s/60);
        const ss = s%60;
        document.getElementById('timer-val').innerText = `${String(m).padStart(2,'0')}:${String(ss).padStart(2,'0')}`;
        if(s<=0) { clearInterval(timer); alert("タイムアップです！"); }
    }, 1000);
}
function stopTimer() { clearInterval(timer); document.getElementById('timer-val').innerText = "00:00"; }

function switchTab(screenId, btn) {
    document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
    document.getElementById('screen-' + screenId).classList.add('active');
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
}

// --- 5. 保存 & 検索 ---
function saveFinalRecord() {
    const name = document.getElementById('cust-name').value;
    if(!name) return alert("客名を入力してください");
    
    const records = JSON.parse(localStorage.getItem('records_ai') || '[]');
    records.push({
        name,
        date: new Date().toLocaleDateString(),
        level: state.currentLevel,
        brand: state.brand,
        img: state.capturedPhoto
    });
    localStorage.setItem('records_ai', JSON.stringify(records));
    alert("カルテを保存しました");
}

function searchRecords() {
    const q = document.getElementById('search-query').value;
    const records = JSON.parse(localStorage.getItem('records_ai') || '[]');
    const html = records.filter(r => r.name.includes(q)).reverse().map(r => `
        <div class="card">
            <strong>${r.date} / ${r.name}</strong><br>
            ランク: ${r.level} (${r.brand})
            ${r.img ? `<img src="${r.img}" style="width:100%; border-radius:5px; margin-top:5px;">` : ''}
        </div>
    `).join('');
    document.getElementById('history-container').innerHTML = html || "履歴がありません";
}
